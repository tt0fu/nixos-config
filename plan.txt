```
# ./modules/misc/foobar.nix
{
  os = {pkgs, inputs, settings ...}:
  {
    # regular nixos configuration, like in configuration.nix
    services.foo = {
      enable = true;
      package = inputs.foo.packages.${pkgs.stdenv.hostPlatform.system}.default;
    };
    users.users.${settings.username}.extraGroups = [ "foobar" ]; # references settings.nix
  };
  home = {pkgs, inputs, style, ...}:
  {
    # home manager configuration, like in home.nix
    programs.bar = {
      enable = true;
      package = inputs.bar.packages.${pkgs.stdenv.hostPlatform.system}.default;
      fontSize = style.fontSize; # references style.nix
    };
  };
}
```

# systems.nix
{
  my-laptop = { # the hostname of my laptop

    ];
    settings = {
      monitor = "eDP-1"
    };
  };
  my-pc = { # the hostname of my pc
    system = "x86_64-linux";
